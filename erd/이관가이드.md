### 4단계: 회원 이관 (AI자세분석)
```sql
INSERT INTO member (id, franchise_id, name, phone, gender, birth_date, height, weight, memo, consultation_enabled, status, created_at)
SELECT
  m.num,
  1,
  m.uname,
  COALESCE(NULLIF(m.sphone, ''), NULLIF(m.mphone, ''), ''),
  CASE m.sex WHEN '남' THEN 'M' WHEN '여' THEN 'F' ELSE NULL END,
  CASE
    WHEN m.uyear != '' AND m.umonth != '' AND m.uday != ''
    THEN STR_TO_DATE(CONCAT(m.uyear, '-', LPAD(m.umonth,2,'0'), '-', LPAD(m.uday,2,'0')), '%Y-%m-%d')
    ELSE NULL
  END,
  NULLIF(mf.uheight, '0'),
  NULLIF(mf.uweight, '0'),
  NULLIF(m.comment, ''),
  0, -- consultation_enabled: 이관 시 기본 비활성, 가맹점에서 개별 설정
  CASE m.bigo
    WHEN '입관' THEN 'ACTIVE'
    WHEN '신규' THEN 'ACTIVE'
    WHEN '일반' THEN 'ACTIVE'
    WHEN '일반신규' THEN 'ACTIVE'
    WHEN '복관' THEN 'ACTIVE'
    WHEN '휴관' THEN 'PAUSED'
    WHEN '명예' THEN 'HONORARY'
    WHEN '퇴관' THEN 'WITHDRAWN'
    ELSE 'ACTIVE'
  END,
  FROM_UNIXTIME(m.signdate)
FROM 기존DB.members m
LEFT JOIN (
  SELECT members_num, uheight, uweight,
         ROW_NUMBER() OVER (PARTITION BY members_num ORDER BY signdate DESC) rn
  FROM 기존DB.members_footprint
) mf ON mf.members_num = m.num AND mf.rn = 1;
```
